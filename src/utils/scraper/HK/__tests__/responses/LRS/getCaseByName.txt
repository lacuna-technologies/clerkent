function get_cookie ( cookie_name )
{
	var str = '(^|;)?' + cookie_name + '=([^;]*)(;|$)';
	var regexp = new RegExp(str, 'g');
	var matches = document.cookie.match(regexp);
	
	if (matches)
	{
		var results = matches[matches.length-1].match(str);
		if ( results )
		return ( unescape ( results[2] ) );
	}
	
	return null;
}
function set_cookie ( name, value, exp_y, exp_m, exp_d, path, domain, secure )
{
	var cookie_string = name + "=" + escape ( value );
	if ( exp_y )
	{
	var expires = new Date ( exp_y, exp_m, exp_d );
	cookie_string += "; expires=" + expires.toGMTString();
	}
	if ( path )
	cookie_string += "; path=" + escape ( path );
	if ( domain )
	cookie_string += "; domain=" + escape ( domain );
	if ( secure )
	cookie_string += "; secure";
	document.cookie = cookie_string;
}

function havevalue(selectBox, nameToSelect){
    var i=false;
    for (var count = 0; count < selectBox.options.length; count++) {          
        if(selectBox[count].selected==true){
          if(selectBox[count].value == nameToSelect) {
            i = true;
          }
        }
        
    }
    return i;
}
